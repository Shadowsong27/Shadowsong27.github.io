Design rationale recorded in project memory (era-taxonomy). */}} {{- $cfg := site.Params.era | default dict -}} {{- $technical := $cfg.technicalCategories | default (slice "data-engineering-log" "data-analysis-log" "devops-log" "toy-tip" "ai-log") -}} {{- $cats := .Params.categories | default slice -}} {{- $isTech := false -}} {{- range $cats -}}{{- if in $technical . -}}{{- $isTech = true -}}{{- end -}}{{- end -}} {{- if $isTech -}} {{- $era := .Params.era -}} {{- if not $era -}} {{- $vibe := time.AsTime ($cfg.vibeCodingStart | default "2025-02-24") -}} {{- $agentic := time.AsTime ($cfg.agenticStart | default "2025-10-01") -}} {{- if ge .Date $agentic -}}{{- $era = "agentic" -}} {{- else if ge .Date $vibe -}}{{- $era = "vibe-coding" -}} {{- else -}}{{- $era = "classic" -}}{{- end -}} {{- end -}} {{- $year := .Date.Year -}} {{- $icon := "" }}{{ $label := "" }}{{ $blurb := "" -}} {{- if eq $era "classic" -}} {{- $icon = "📜" }}{{ $label = "Pre-AI era" -}} {{- $blurb = printf "Written in %d, before AI-assisted coding.
